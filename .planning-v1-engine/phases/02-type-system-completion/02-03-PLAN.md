**Step 13: Extend parse_postfix_expression (line ~5915)**

Replace the current `while self.match_punctuation(CypherPunctuation::Dot)` loop with a more general loop that handles both `.property` and `[subscript]`:

```rust
fn parse_postfix_expression(&mut self) -> Result<CypherExpression, ParseError> {
    let mut expr = self.parse_primary_expression()?;
    loop {
        if self.match_punctuation(CypherPunctuation::Dot) {
            let property = self.parse_identifier("for property access")?;
            expr = CypherExpression::PropertyAccess {
                target: Box::new(expr),
                property,
            };
        } else if self.match_punctuation(CypherPunctuation::LeftBracket) {
            // Parse subscript: [index] or [start..end]
            if self.match_punctuation(CypherPunctuation::RightBracket) {
                // Empty subscript [] - return empty list
                return Err(self.error_here("expected index expression in subscript"));
            }
            let index = self.parse_expression()?;
            let end = if self.match_punctuation(CypherPunctuation::Range) {
                // Range: [start..end]
                let end_expr = self.parse_expression()?;
                Some(Box::new(end_expr))
            } else {
                None
            };
            self.expect_punctuation(CypherPunctuation::RightBracket)?;
            expr = CypherExpression::Subscript {
                base: Box::new(expr),
                index: Box::new(index),
                end,
            };
        } else {
            break;
        }
    }
    Ok(expr)
}
```

The parser already has `match_punctuation` and `expect_punctuation` methods, and the `Range` variant exists. However, verify that the parser helper methods (`match_punctuation`, `check_punctuation`) handle the `Range` variant correctly. They should, since they pattern-match on `CypherPunctuation`.

**Step 2: Add Subscript evaluator**

Add a new match arm in `evaluate_expression` for the Subscript variant. Place it near the other expression evaluators (around the ListLiteral arm):

```rust
CypherExpression::Subscript { base, index, end } => {
    let base_val = self.evaluate_expression(base, row, snapshot_txn_id)?;
    let index_val = self.evaluate_expression(index, row, snapshot_txn_id)?;

    match (&base_val, end) {
        // Single index access: list[n]
        (RuntimeValue::Property(PropertyValue::List(items)), None) => {
            let idx = match &index_val {
                RuntimeValue::Property(PropertyValue::I64(i)) => *i,
                _ => return Ok(RuntimeValue::Null),
            };
            // Support negative indexing: list[-1] = last element
            let resolved_idx = if idx < 0 {
                (items.len() as i64 + idx) as usize
            } else {
                idx as usize
            };
            Ok(items.get(resolved_idx)
                .map(|item| RuntimeValue::Property(item.clone()))
                .unwrap_or(RuntimeValue::Null))
        }
        // Range slice: list[start..end]
        (RuntimeValue::Property(PropertyValue::List(items)), Some(end_expr)) => {
            let start_idx = match &index_val {
                RuntimeValue::Property(PropertyValue::I64(i)) => *i as usize,
                _ => return Ok(RuntimeValue::Null),
            };
            let end_val = self.evaluate_expression(end_expr, row, snapshot_txn_id)?;
            let end_idx = match &end_val {
                RuntimeValue::Property(PropertyValue::I64(i)) => *i as usize,
                _ => return Ok(RuntimeValue::Null),
            };
            let start = start_idx.min(items.len());
            let end = end_idx.min(items.len());
            if start >= end {
                Ok(RuntimeValue::Property(PropertyValue::List(Vec::new())))
            } else {
                Ok(RuntimeValue::Property(PropertyValue::List(
                    items[start..end].to_vec()
                )))
            }
        }
        // String subscript: str[n]
        (RuntimeValue::Property(PropertyValue::String(s)), None) => {
            let idx = match &index_val {
                RuntimeValue::Property(PropertyValue::I64(i)) => *i,
                _ => return Ok(RuntimeValue::Null),
            };
            let resolved_idx = if idx < 0 {
                (s.len() as i64 + idx) as usize
            } else {
                idx as usize
            };
            Ok(s.chars().nth(resolved_idx)
                .map(|ch| RuntimeValue::Property(PropertyValue::String(ch.to_string())))
                .unwrap_or(RuntimeValue::Null))
        }
        _ => Ok(RuntimeValue::Null),
    }
}
```

**Step 3: Implement ListComprehension evaluator (line ~12888)**

Replace the current stub `CypherExpression::ListComprehension { .. } => Ok(RuntimeValue::Null)` with:

```rust
CypherExpression::ListComprehension {
    variable,
    list,
    predicate,
    projection,
} => {
    let list_val = self.evaluate_expression(list, row, snapshot_txn_id)?;
    let items = match list_val {
        RuntimeValue::Property(PropertyValue::List(items)) => items,
        _ => return Ok(RuntimeValue::Property(PropertyValue::List(Vec::new()))),
    };

    let mut result = Vec::new();
    for item in &items {
        // Create a modified row with the variable bound to the current item
        let mut local_row = row.clone();
        // Find or add column for the variable
        let col_idx = local_row.columns.iter().position(|c| c == variable)
            .unwrap_or_else(|| {
                local_row.columns.push(variable.clone());
                local_row.values.push(RuntimeValue::Null);
                local_row.columns.len() - 1
            });
        local_row.values[col_idx] = RuntimeValue::Property(item.clone());

        // Check predicate if present
        if let Some(pred) = predicate {
            let pred_val = self.evaluate_expression(pred, &local_row, snapshot_txn_id)?;
            if !runtime_value_truthy(&pred_val) {
                continue;
            }
        }

        // Apply projection if present, otherwise use the item itself
        let projected = if let Some(proj) = projection {
            self.evaluate_expression(proj, &local_row, snapshot_txn_id)?
        } else {
            RuntimeValue::Property(item.clone())
        };
        result.push(runtime_to_property_value(&projected));
    }

    Ok(RuntimeValue::Property(PropertyValue::List(result)))
}
```
