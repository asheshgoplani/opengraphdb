```rust
#[cfg(test)]
mod budget_gates {
    use ogdb_core::{Database, Header};
    use std::time::Instant;
    use tempfile::TempDir;

    /// Cross-platform RSS measurement using `ps -o rss=`.
    /// Works on macOS and Linux without extra dependencies.
    /// Returns resident set size in bytes.
    fn get_rss_bytes() -> u64 {
        let output = std::process::Command::new("ps")
            .args(["-o", "rss=", "-p", &std::process::id().to_string()])
            .output()
            .ok();
        output
            .and_then(|o| String::from_utf8(o.stdout).ok())
            .and_then(|s| s.trim().parse::<u64>().ok())
            .map(|kb| kb * 1024) // ps reports in KB
            .unwrap_or(0)
    }

    /// Measure total on-disk size of all files in a directory (non-recursive).
    fn dir_disk_bytes(dir: &std::path::Path) -> u64 {
        std::fs::read_dir(dir)
            .map(|entries| {
                entries
                    .filter_map(|e| e.ok())
                    .filter_map(|e| e.metadata().ok().map(|m| m.len()))
                    .sum()
            })
            .unwrap_or(0)
    }

    const BATCH_SIZE: u64 = 50_000;

    /// Build a graph with the given node and edge counts using batched transactions.
    /// Returns (TempDir, Database) — TempDir must be kept alive to prevent cleanup.
    fn build_budget_graph(tag: &str, node_count: u64, edge_count: u64) -> (TempDir, Database) {
        let dir = TempDir::new().expect("temp dir");
        let db_path = dir.path().join(format!("{tag}.ogdb"));
        let mut db = Database::init(&db_path, Header::default_v1()).expect("init db");

        // Create nodes in batches
        let mut created = 0u64;
        while created < node_count {
            let batch = BATCH_SIZE.min(node_count - created);
            let mut tx = db.begin_write();
            for _ in 0..batch {
                tx.create_node().expect("create node");
            }
            tx.commit().expect("commit node batch");
            created += batch;
        }
        assert_eq!(db.node_count(), node_count);

        // Create edges in batches using deterministic pseudo-random endpoints.
        // Uses xorshift64* for reproducibility without extra dependencies.
        let mut rng_state: u64 = 42;
        let mut edges_created = 0u64;
        while edges_created < edge_count {
            let batch = BATCH_SIZE.min(edge_count - edges_created);
            let mut tx = db.begin_write();
            for _ in 0..batch {
                // xorshift64* PRNG
                rng_state ^= rng_state >> 12;
                rng_state ^= rng_state << 25;
                rng_state ^= rng_state >> 27;
                let r = rng_state.wrapping_mul(0x2545F4914F6CDD1D);
                let src = r % node_count;
                rng_state ^= rng_state >> 12;
                rng_state ^= rng_state << 25;
                rng_state ^= rng_state >> 27;
                let r2 = rng_state.wrapping_mul(0x2545F4914F6CDD1D);
                let dst = r2 % node_count;
                tx.add_edge(src, dst).expect("add edge");
            }
            tx.commit().expect("commit edge batch");
            edges_created += batch;
        }

        // Checkpoint to flush WAL to main file
        db.checkpoint().expect("checkpoint");

        (dir, db)
    }

    // --- Smoke test (runs in normal CI) ---

    #[test]
    fn budget_measurement_smoke_test() {
        let node_count = 1_000;
        let edge_count = 5_000;
        let (dir, db) = build_budget_graph("smoke", node_count, edge_count);

        assert_eq!(db.node_count(), node_count);
        assert_eq!(db.edge_count(), edge_count);

        let rss = get_rss_bytes();
        assert!(rss > 0, "RSS measurement returned zero; ps may not be available");

        let disk = dir_disk_bytes(dir.path());
        assert!(disk > 0, "disk measurement returned zero");

        eprintln!(
            "smoke test: nodes={}, edges={}, rss={:.2} MB, disk={:.2} MB",
            node_count,
            edge_count,
            rss as f64 / (1024.0 * 1024.0),
            disk as f64 / (1024.0 * 1024.0),
        );
    }

    // --- Full-scale gate tests (run on dedicated hardware) ---

    const GATE_NODE_COUNT: u64 = 1_000_000;
    const GATE_EDGE_COUNT: u64 = 5_000_000;
    const MEMORY_BUDGET_BYTES: u64 = 500 * 1024 * 1024; // 500 MB
    const DISK_BUDGET_BYTES: u64 = 1_000 * 1024 * 1024; // 1 GB

    #[test]
    #[ignore = "budget gate; run on dedicated hardware with sufficient memory"]
    fn memory_budget_gate_1m_nodes_5m_edges() {
        eprintln!("building budget graph: {} nodes, {} edges ...", GATE_NODE_COUNT, GATE_EDGE_COUNT);
        let started = Instant::now();
        let (_dir, _db) = build_budget_graph("mem-budget", GATE_NODE_COUNT, GATE_EDGE_COUNT);
        let build_secs = started.elapsed().as_secs_f64();
        eprintln!("graph built in {build_secs:.1}s");

        let rss = get_rss_bytes();
        let rss_mb = rss as f64 / (1024.0 * 1024.0);
        let budget_mb = MEMORY_BUDGET_BYTES as f64 / (1024.0 * 1024.0);
        eprintln!("RSS: {rss_mb:.2} MB (budget: {budget_mb:.0} MB)");

        assert!(
            rss > 0,
            "RSS measurement returned zero; ps may not be available on this platform"
        );
        assert!(
            rss < MEMORY_BUDGET_BYTES,
            "QUAL-01 FAILED: RSS {rss_mb:.2} MB exceeds {budget_mb:.0} MB budget"
        );
    }

    #[test]
    #[ignore = "budget gate; run on dedicated hardware with sufficient memory"]
    fn disk_budget_gate_1m_nodes_5m_edges() {
        eprintln!("building budget graph: {} nodes, {} edges ...", GATE_NODE_COUNT, GATE_EDGE_COUNT);
        let started = Instant::now();
        let (dir, mut db) = build_budget_graph("disk-budget", GATE_NODE_COUNT, GATE_EDGE_COUNT);
        let build_secs = started.elapsed().as_secs_f64();
        eprintln!("graph built in {build_secs:.1}s");

        // Ensure everything is flushed before measuring
        db.checkpoint().expect("final checkpoint");

        let disk = dir_disk_bytes(dir.path());
        let disk_mb = disk as f64 / (1024.0 * 1024.0);
        let budget_mb = DISK_BUDGET_BYTES as f64 / (1024.0 * 1024.0);
        eprintln!("disk: {disk_mb:.2} MB (budget: {budget_mb:.0} MB)");

        // List individual files for diagnostic output
        if let Ok(entries) = std::fs::read_dir(dir.path()) {
            for entry in entries.flatten() {
                if let Ok(meta) = entry.metadata() {
                    let name = entry.file_name();
                    let size_mb = meta.len() as f64 / (1024.0 * 1024.0);
                    eprintln!("  {}: {size_mb:.2} MB", name.to_string_lossy());
                }
            }
        }

        assert!(
            disk > 0,
            "disk measurement returned zero"
        );
        assert!(
            disk < DISK_BUDGET_BYTES,
            "QUAL-02 FAILED: disk {disk_mb:.2} MB exceeds {budget_mb:.0} MB budget"
        );
    }
}
```

**IMPORTANT NOTES:**

1. The `build_budget_graph` helper is shared between both gate tests. Each test calls it independently so they can run in isolation or in parallel.

2. The `get_rss_bytes()` function uses `ps -o rss=` which works on both macOS and Linux. It spawns a child process, which is negligible overhead for a gate benchmark.

3. The `dir_disk_bytes()` function measures all files in the temp directory, covering both the `.ogdb` main file and the `.ogdb-wal` WAL file.

4. The smoke test (`budget_measurement_smoke_test`) runs in normal CI to ensure the measurement machinery works. It uses 1K nodes and 5K edges, finishing in under a second.

5. The full-scale tests are `#[ignore]` so they only run when explicitly requested (`cargo test -p ogdb-bench -- --ignored`).

6. Edge generation uses an inline xorshift64* PRNG (same algorithm as the existing `DetRng` in the file) rather than importing it, since `DetRng` is defined in the binary module and not accessible from the test module. Alternatively, if you prefer, you can move `DetRng` to a shared location, but the inline approach is simpler and avoids modifying existing code.
