<objective>
Add worst-case optimal join (WCOJ) strategy to the OpenGraphDB query planner and executor.

Purpose: QOPT-01 requires the query planner to select WCOJ for multi-way pattern queries when estimated cardinality favors it over binary joins. The current planner always generates a left-deep tree of binary Expand operators, which can produce intermediate results exponentially larger than the final output on cyclic or high-fan-out patterns.

Output: `PhysicalWcojJoin` plan variant, WCOJ candidate detection in the physical planner, cost-based selection between WCOJ and binary chains, and a leapfrog-intersection executor that evaluates all relations simultaneously.
</objective>

/// Analyze a logical plan tree to detect whether it is a chain of Expand operators
/// rooted at a Scan, with 2+ Expand steps, suitable for WCOJ.
///
/// Returns Some(candidate) if the pattern has 3+ connected variables (2+ Expand steps)
/// and at least one cycle or 3+ expansions making WCOJ potentially beneficial.
fn detect_wcoj_candidate(plan: &LogicalPlan) -> Option<WcojCandidate> {
    // Walk the plan tree to extract a chain of Expand operators rooted at a Scan.
    let mut expands = Vec::<(String, Option<String>, RelationshipDirection, Option<String>, Option<String>, Option<TemporalFilter>)>::new();
    let mut current = plan;

    // Peel off any top-level Filter (WHERE clause applied after expansions).
    // We don't include the filter in the WCOJ candidate; it will be applied after.
    loop {
        match current {
            LogicalPlan::Expand {
                input,
                from,
                edge_type,
                direction,
                to,
                edge_variable,
                temporal_filter,
            } => {
                expands.push((
                    from.clone(),
                    edge_type.clone(),
                    *direction,
                    to.clone(),
                    edge_variable.clone(),
                    temporal_filter.clone(),
                ));
                current = input;
            }
            _ => break,
        }
    }

    // Need at least 2 Expand steps (3+ variables) for WCOJ to be beneficial.
    if expands.len() < 2 {
        return None;
    }

    // The base must be a Scan (possibly wrapped in a Filter for node property predicates).
    let (base_label, base_variable) = match current {
        LogicalPlan::Scan { label, variable } => (label.clone(), variable.clone()),
        LogicalPlan::Filter { input, .. } => match input.as_ref() {
            LogicalPlan::Scan { label, variable } => (label.clone(), variable.clone()),
            _ => return None,
        },
        _ => return None,
    };

    // Reverse expands since we collected them from outermost to innermost.
    expands.reverse();

    // Build variable order and relations.
    let mut variable_order = Vec::<String>::new();
    let mut output_columns = Vec::<String>::new();
    let mut seen_variables = std::collections::BTreeSet::<String>::new();

    // Add base variable.
    let base_var = base_variable.clone().unwrap_or_else(|| "_anon".to_string());
    variable_order.push(base_var.clone());
    output_columns.push(base_var.clone());
    seen_variables.insert(base_var);

    let mut relations = Vec::<WcojRelation>::new();
    let mut is_cyclic = false;

    for (from, edge_type, direction, to, edge_variable, temporal_filter) in &expands {
        let to_name = to.clone().unwrap_or_else(|| format!("_anon_{}", relations.len()));

        if seen_variables.contains(&to_name) {
            is_cyclic = true;
        } else {
            variable_order.push(to_name.clone());
            seen_variables.insert(to_name.clone());
        }

        if !output_columns.contains(&to_name) {
            output_columns.push(to_name.clone());
        }
        if let Some(ev) = edge_variable {
            if !output_columns.contains(ev) {
                output_columns.push(ev.clone());
            }
        }

        relations.push(WcojRelation {
            from: from.clone(),
            to: to_name,
            edge_type: edge_type.clone(),
            direction: *direction,
            edge_variable: edge_variable.clone(),
            temporal_filter: temporal_filter.clone(),
        });
    }

    // Only beneficial for cyclic patterns or 3+ expansions (4+ variables).
    if !is_cyclic && expands.len() < 3 {
        return None;
    }

    Some(WcojCandidate {
        base_label,
        base_variable,
        relations,
        variable_order,
        output_columns,
        is_cyclic,
    })
}
```

**Step 3: Add estimate_wcoj_cost() function**

Add this function after `detect_wcoj_candidate`:

/// Estimate the cost and output rows for a WCOJ execution.
///
/// WCOJ cost model:
/// - For k relations over n variables, the AGM bound gives output ≤ |DB|^(k/2)
/// - Build cost: sum of sorted adjacency list construction = sum of |R_i|
/// - Intersection cost: for each value of var[0], intersect sorted neighbor lists
///   Cost ≈ sum(|R_i|) * log(max(|R_i|)) for leapfrog merge
fn estimate_wcoj_cost(db: &Database, candidate: &WcojCandidate) -> (u64, f64) {
    let mut total_relation_size: u64 = 0;
    let mut max_relation_size: u64 = 0;

    for relation in &candidate.relations {
        let size = edge_count_for_type(db, relation.edge_type.as_deref());
        total_relation_size = total_relation_size.saturating_add(size);
        max_relation_size = max_relation_size.max(size);
    }

    let num_relations = candidate.relations.len() as f64;
    let node_count = db.node_count().max(1) as f64;

    // AGM bound: for k relations, output bounded by |DB|^(k/2).
    // Use a tighter estimate: geometric mean of relation sizes.
    let geo_mean = if candidate.relations.is_empty() {
        1.0
    } else {
        let log_sum: f64 = candidate.relations.iter()
            .map(|r| (edge_count_for_type(db, r.edge_type.as_deref()).max(1) as f64).ln())
            .sum();
        (log_sum / num_relations).exp()
    };

    // For cyclic patterns, output is typically much smaller than binary join intermediate.
    let estimated_rows = if candidate.is_cyclic {
        // Triangle-like: estimate as geo_mean^(3/2) / node_count
        ((geo_mean.powf(1.5) / node_count).ceil() as u64).max(1)
    } else {
        // Acyclic multi-way: estimate as product of avg degrees
        let mut rows = label_cardinality_for_scan(db, candidate.base_label.as_deref()).max(1);
        for relation in &candidate.relations {
            let edge_count = edge_count_for_type(db, relation.edge_type.as_deref()).max(1) as f64;
            let avg_degree = edge_count / node_count;
            rows = ((rows as f64) * avg_degree.max(1.0)).ceil() as u64;
        }
        rows
    };

    // Cost: build sorted indexes + leapfrog intersection.
    // Build: sum of relation sizes (one pass each).
    // Intersection: for each candidate tuple prefix, log-factor intersection.
    let build_cost = total_relation_size as f64;
    let intersection_cost = estimated_rows as f64 * (max_relation_size as f64).ln().max(1.0) * num_relations;
    let total_cost = build_cost + intersection_cost;

    (estimated_rows, total_cost)
}
```

/// Estimate the cost of the standard binary Expand chain for comparison with WCOJ.
fn estimate_binary_chain_cost(db: &Database, candidate: &WcojCandidate) -> (u64, f64) {
    let node_count = db.node_count().max(1);
    let base_rows = label_cardinality_for_scan(db, candidate.base_label.as_deref()).max(1);

    let mut current_rows = base_rows;
    let mut total_cost = base_rows as f64; // Scan cost.

    for relation in &candidate.relations {
        let edge_rows = edge_count_for_type(db, relation.edge_type.as_deref());
        let avg_degree = edge_rows as f64 / node_count as f64;
        let next_rows = ((current_rows as f64) * avg_degree.max(1.0)).ceil() as u64;

        // HashJoin cost for this expansion.
        let expand_cost = (edge_rows + current_rows) as f64;
        total_cost += expand_cost;
        current_rows = next_rows;
    }

    (current_rows, total_cost)
}

```rust
    /// Execute a WCOJ join using leapfrog-style intersection.
    ///
    /// Algorithm:
    /// 1. Build sorted adjacency indexes for each relation.
    /// 2. For each value of the first variable (from input scan):
    ///    a. Collect candidate sets for the next variable from all relevant relations.
    ///    b. Intersect candidate sets (sorted intersection).
    ///    c. For each surviving candidate, recurse to the next variable.
    /// 3. Output rows for complete tuples that satisfy all relations.
    fn execute_wcoj_join(
        &self,
        input_batches: &[RuntimeBatch],
        relations: &[WcojRelation],
        variable_order: &[String],
        output_columns: &[String],
        snapshot_txn_id: u64,
    ) -> Result<Vec<RuntimeBatch>, QueryError> {
        // Step 1: Build sorted adjacency indexes for each relation.
        // For each relation, we build: HashMap<u64, Vec<(u64, RuntimeEdgeRef)>>
        // keyed by the "from" node, with sorted neighbor lists.
        let mut relation_indexes: Vec<HashMap<u64, Vec<(u64, RuntimeEdgeRef)>>> = Vec::new();

        for relation in relations {
            let mut index = HashMap::<u64, Vec<(u64, RuntimeEdgeRef)>>::new();
            for (edge_id, record) in self.edge_records.iter().enumerate() {
                let edge_id = edge_id as u64;
                if !self.is_edge_visible_at(edge_id, snapshot_txn_id) {
                    continue;
                }
                if !self.edge_matches_temporal_filter(
                    edge_id,
                    relation.temporal_filter.as_ref(),
                    snapshot_txn_id,
                )? {
                    continue;
                }
                if !self.is_node_visible_at(record.src, snapshot_txn_id)
                    || !self.is_node_visible_at(record.dst, snapshot_txn_id)
                {
                    continue;
                }
                let edge_type_value = self.edge_type_at(edge_id, snapshot_txn_id)?;
                if let Some(ref expected) = relation.edge_type {
                    if edge_type_value.as_deref() != Some(expected.as_str()) {
                        continue;
                    }
                }
                let edge_ref = RuntimeEdgeRef {
                    edge_id,
                    src: record.src,
                    dst: record.dst,
                    edge_type: edge_type_value,
                };
                match relation.direction {
                    RelationshipDirection::LeftToRight => {
                        index
                            .entry(record.src)
                            .or_default()
                            .push((record.dst, edge_ref));
                    }
                    RelationshipDirection::RightToLeft => {
                        index
                            .entry(record.dst)
                            .or_default()
                            .push((record.src, edge_ref));
                    }
                    RelationshipDirection::Undirected => {
                        index
                            .entry(record.src)
                            .or_default()
                            .push((record.dst, edge_ref.clone()));
                        index
                            .entry(record.dst)
                            .or_default()
                            .push((record.src, edge_ref));
                    }
                }
            }
            // Sort neighbor lists for each key (enables intersection).
            for neighbors in index.values_mut() {
                neighbors.sort_by_key(|(node_id, _)| *node_id);
            }
            relation_indexes.push(index);
        }

        // Step 2: Build mapping from (from_var, to_var) -> relation index.
        let mut var_to_relation: HashMap<(String, String), Vec<usize>> = HashMap::new();
        // Also build reverse mapping for back-edges in cyclic patterns.
        let mut var_to_reverse_relation: HashMap<(String, String), Vec<usize>> = HashMap::new();
        for (idx, relation) in relations.iter().enumerate() {
            var_to_relation
                .entry((relation.from.clone(), relation.to.clone()))
                .or_default()
                .push(idx);
            var_to_reverse_relation
                .entry((relation.to.clone(), relation.from.clone()))
                .or_default()
                .push(idx);
        }

        // Step 3: Execute using recursive variable-at-a-time processing.
        let input_rows = rows_from_batches(input_batches);
        let mut result_rows = Vec::<BTreeMap<String, RuntimeValue>>::new();

        for input_row in &input_rows {
            // Current bindings: start with input row bindings.
            let mut bindings = input_row.clone();

            self.wcoj_recurse(
                &variable_order,
                1, // Start from variable index 1 (index 0 is already bound from input).
                &mut bindings,
                &relation_indexes,
                relations,
                &var_to_relation,
                &var_to_reverse_relation,
                output_columns,
                snapshot_txn_id,
                &mut result_rows,
            )?;
        }

        Ok(batches_from_rows(&result_rows, output_columns, QUERY_BATCH_SIZE))
    }

    /// Recursive helper for WCOJ: bind the next variable by intersecting candidate sets
    /// from all relevant relations.
    fn wcoj_recurse(
        &self,
        variable_order: &[String],
        var_idx: usize,
        bindings: &mut BTreeMap<String, RuntimeValue>,
        relation_indexes: &[HashMap<u64, Vec<(u64, RuntimeEdgeRef)>>],
        relations: &[WcojRelation],
        var_to_relation: &HashMap<(String, String), Vec<usize>>,
        var_to_reverse_relation: &HashMap<(String, String), Vec<usize>>,
        output_columns: &[String],
        snapshot_txn_id: u64,
        result_rows: &mut Vec<BTreeMap<String, RuntimeValue>>,
    ) -> Result<(), QueryError> {
        // Base case: all variables are bound, emit a result row.
        if var_idx >= variable_order.len() {
            // Check that all relations are satisfied (validation for cyclic patterns).
            // For cyclic patterns, some relations connect already-bound variables.
            for (rel_idx, relation) in relations.iter().enumerate() {
                let from_id = match bindings.get(&relation.from) {
                    Some(RuntimeValue::Node(id)) => *id,
                    _ => continue,
                };
                let to_id = match bindings.get(&relation.to) {
                    Some(RuntimeValue::Node(id)) => *id,
                    _ => continue,
                };
                // Verify the edge exists in the relation index.
                if let Some(neighbors) = relation_indexes[rel_idx].get(&from_id) {
                    if !neighbors.iter().any(|(dst, _)| *dst == to_id) {
                        return Ok(()); // This tuple doesn't satisfy all relations.
                    }
                } else {
                    return Ok(());
                }
            }

            // Emit result row with edge variables.
            let mut row = BTreeMap::<String, RuntimeValue>::new();
            for col in output_columns {
                if let Some(val) = bindings.get(col) {
                    row.insert(col.clone(), val.clone());
                }
            }

            // Add edge variables by looking up edges.
            for (rel_idx, relation) in relations.iter().enumerate() {
                if let Some(ref ev) = relation.edge_variable {
                    if output_columns.contains(ev) && !row.contains_key(ev) {
                        let from_id = match bindings.get(&relation.from) {
                            Some(RuntimeValue::Node(id)) => *id,
                            _ => continue,
                        };
                        let to_id = match bindings.get(&relation.to) {
                            Some(RuntimeValue::Node(id)) => *id,
                            _ => continue,
                        };
                        if let Some(neighbors) = relation_indexes[rel_idx].get(&from_id) {
                            if let Some((_, edge_ref)) = neighbors.iter().find(|(dst, _)| *dst == to_id) {
                                row.insert(ev.clone(), RuntimeValue::Edge(edge_ref.clone()));
                            }
                        }
                    }
                }
            }

            result_rows.push(row);
            return Ok(());
        }

        let target_var = &variable_order[var_idx];

        // Check if this variable is already bound (cyclic case).
        if bindings.contains_key(target_var) {
            // Variable already bound; just recurse to validate remaining variables.
            return self.wcoj_recurse(
                variable_order,
                var_idx + 1,
                bindings,
                relation_indexes,
                relations,
                var_to_relation,
                var_to_reverse_relation,
                output_columns,
                snapshot_txn_id,
                result_rows,
            );
        }

        // Collect candidate sets from all relations that connect a bound variable to target_var.
        let mut candidate_sets: Vec<Vec<u64>> = Vec::new();

        // Check forward relations: (bound_var -> target_var).
        for (bound_var, _) in bindings.iter() {
            let key = (bound_var.clone(), target_var.clone());
            if let Some(rel_indices) = var_to_relation.get(&key) {
                for &rel_idx in rel_indices {
                    let from_id = match bindings.get(bound_var) {
                        Some(RuntimeValue::Node(id)) => *id,
                        _ => continue,
                    };
                    if let Some(neighbors) = relation_indexes[rel_idx].get(&from_id) {
                        let candidates: Vec<u64> = neighbors.iter().map(|(id, _)| *id).collect();
                        candidate_sets.push(candidates);
                    } else {
                        // No neighbors: intersection will be empty.
                        return Ok(());
                    }
                }
            }

            // Check reverse relations: relation says (target_var -> bound_var),
            // so we need to find nodes that reach bound_var.
            let rev_key = (bound_var.clone(), target_var.clone());
            if let Some(rel_indices) = var_to_reverse_relation.get(&rev_key) {
                for &rel_idx in rel_indices {
                    // The relation goes from target_var to bound_var.
                    // We need nodes in relation_indexes[rel_idx] whose neighbors include bound_var_id.
                    // This is expensive without a reverse index, so build one on the fly.
                    let bound_id = match bindings.get(bound_var) {
                        Some(RuntimeValue::Node(id)) => *id,
                        _ => continue,
                    };
                    let mut candidates = Vec::<u64>::new();
                    for (src, neighbors) in &relation_indexes[rel_idx] {
                        if neighbors.iter().any(|(dst, _)| *dst == bound_id) {
                            candidates.push(*src);
                        }
                    }
                    candidates.sort();
                    if candidates.is_empty() {
                        return Ok(());
                    }
                    candidate_sets.push(candidates);
                }
            }
        }

        if candidate_sets.is_empty() {
            // No relations constrain this variable; it's free (shouldn't happen in well-formed patterns).
            return Ok(());
        }

        // Intersect all candidate sets using sorted merge (leapfrog style).
        let intersection = if candidate_sets.len() == 1 {
            candidate_sets.into_iter().next().unwrap()
        } else {
            // Sort each set and intersect pairwise.
            let mut result = candidate_sets[0].clone();
            result.sort();
            result.dedup();
            for set in &candidate_sets[1..] {
                let mut sorted_set = set.clone();
                sorted_set.sort();
                sorted_set.dedup();
                result = sorted_intersect(&result, &sorted_set);
                if result.is_empty() {
                    return Ok(());
                }
            }
            result
        };

        // Recurse for each value in the intersection.
        for node_id in intersection {
            bindings.insert(target_var.clone(), RuntimeValue::Node(node_id));
            self.wcoj_recurse(
                variable_order,
                var_idx + 1,
                bindings,
                relation_indexes,
                relations,
                var_to_relation,
                var_to_reverse_relation,
                output_columns,
                snapshot_txn_id,
                result_rows,
            )?;
            bindings.remove(target_var);
        }

        Ok(())
    }
```

/// Sorted intersection of two sorted, deduplicated u64 slices.
fn sorted_intersect(a: &[u64], b: &[u64]) -> Vec<u64> {
    let mut result = Vec::new();
    let (mut i, mut j) = (0, 0);
    while i < a.len() && j < b.len() {
        match a[i].cmp(&b[j]) {
            std::cmp::Ordering::Less => i += 1,
            std::cmp::Ordering::Greater => j += 1,
            std::cmp::Ordering::Equal => {
                result.push(a[i]);
                i += 1;
                j += 1;
            }
        }
    }
    result
}
