**4. Implement `validate_against_shacl()`:**

```rust
fn validate_against_shacl(
    db: &Database,
    shapes: &[NodeShapeConstraint],
) -> Vec<ShaclViolation> {
    let mut violations = Vec::new();
    let node_count = db.header.node_count();

    for shape in shapes {
        for idx in 0..node_count {
            // Check if node has the target label
            let labels = &db.meta.node_labels[idx];
            if !labels.contains(&shape.target_class) {
                continue;
            }
            // Get current properties for the node
            let props = db.current_node_properties(idx);
            // Check each required property
            for required_prop in &shape.required_properties {
                if !props.contains_key(required_prop) {
                    violations.push(ShaclViolation {
                        node_id: idx as u64,
                        shape_id: shape.shape_id.clone(),
                        target_class: shape.target_class.clone(),
                        violated_property: required_prop.clone(),
                        message: format!(
                            "Node {} (:{}) is missing required property '{}'",
                            idx, shape.target_class, required_prop
                        ),
                    });
                }
            }
        }
    }
    violations
}
```

Note: The method to get current node properties may be `db.meta.node_properties[idx]` or accessed through the node property store. Examine the existing `handle_query` or `process_rdf_quad` patterns to find the correct accessor. The key is to get the PropertyMap for a given node index.

```rust
fn handle_validate_shacl(db_path: &str, shapes_path: &Path) -> Result<(), Box<dyn std::error::Error>> {
    let shapes = parse_shacl_shapes(shapes_path)?;
    if shapes.is_empty() {
        println!("No SHACL NodeShape constraints found in shapes file.");
        return Ok(());
    }

    let db = Database::open(Path::new(db_path))?;
    let violations = validate_against_shacl(&db, &shapes);

    if violations.is_empty() {
        println!("Validation passed: graph conforms to all SHACL shapes.");
        // Exit 0 (implicit)
    } else {
        println!("Validation failed: {} violation(s) found.\n", violations.len());
        for v in &violations {
            println!("  VIOLATION: {}", v.message);
        }
        // Output as JSON for machine consumption
        let json = serde_json::to_string_pretty(&violations)?;
        println!("\n{}", json);
        std::process::exit(1);
    }
    Ok(())
}
```
