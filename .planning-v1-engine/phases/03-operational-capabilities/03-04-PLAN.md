```rust
/// Summary of a mutation query execution.
///
/// Returned by [`Database::execute()`] for write operations like
/// `CREATE`, `SET`, `DELETE`, and `REMOVE`.
#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ExecutionSummary {
    /// Number of rows in the result (if any RETURN clause was present).
    pub rows_returned: usize,
    /// Node count before the query.
    pub nodes_before: u64,
    /// Node count after the query.
    pub nodes_after: u64,
    /// Edge count before the query.
    pub edges_before: u64,
    /// Edge count after the query.
    pub edges_after: u64,
}

impl ExecutionSummary {
    /// Number of nodes created (nodes_after - nodes_before), clamped to 0.
    pub fn nodes_created(&self) -> u64 {
        self.nodes_after.saturating_sub(self.nodes_before)
    }

    /// Number of edges created (edges_after - edges_before), clamped to 0.
    pub fn edges_created(&self) -> u64 {
        self.edges_after.saturating_sub(self.edges_before)
    }
}
```

**Step 3: Add explain() method**

```rust
    /// Return the query execution plan as human-readable text without executing the query.
    ///
    /// This is equivalent to prefixing a query with `EXPLAIN` in Cypher.
    ///
    /// # Example
    /// ```no_run
    /// # use ogdb_core::{Database, Header};
    /// let mut db = Database::init("my.ogdb", Header::default_v1()).unwrap();
    /// let plan_text = db.explain("MATCH (n:Person) WHERE n.name = 'Alice' RETURN n").unwrap();
    /// println!("{plan_text}");
    /// ```
    pub fn explain(&mut self, query: &str) -> Result<String, QueryError> {
        let explain_query = if query.trim().to_ascii_uppercase().starts_with("EXPLAIN ") {
            query.to_string()
        } else {
            format!("EXPLAIN {query}")
        };
        let result = self.query(&explain_query)?;
        // EXPLAIN queries return a single-column result with the plan text
        // Extract the plan text from the result
        Ok(result.to_table())
    }
```

**Step 4: Add execute() method**

```rust
    /// Execute a mutation query and return an execution summary.
    ///
    /// This is a convenience method for write operations (`CREATE`, `SET`, `DELETE`, `REMOVE`).
    /// It calls [`query()`](Database::query) internally and wraps the result with
    /// before/after counts.
    ///
    /// # Example
    /// ```no_run
    /// # use ogdb_core::{Database, Header};
    /// let mut db = Database::init("my.ogdb", Header::default_v1()).unwrap();
    /// let summary = db.execute("CREATE (n:Person {name: 'Alice'})").unwrap();
    /// assert_eq!(summary.nodes_created(), 1);
    /// ```
    pub fn execute(&mut self, query: &str) -> Result<ExecutionSummary, QueryError> {
        let nodes_before = self.node_count();
        let edges_before = self.edge_count();
        let result = self.query(query)?;
        let nodes_after = self.node_count();
        let edges_after = self.edge_count();

        Ok(ExecutionSummary {
            rows_returned: result.row_count(),
            nodes_before,
            nodes_after,
            edges_before,
            edges_after,
        })
    }
```
