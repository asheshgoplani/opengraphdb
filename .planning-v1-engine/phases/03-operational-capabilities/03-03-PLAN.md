#[derive(Debug, Clone, PartialEq, Eq)]
enum MigrationAction {
    AddLabel(String),
    DropLabel(String),
    AddEdgeType(String),
    DropEdgeType(String),
    AddPropertyKey(String),
    DropPropertyKey(String),
    AddIndex { label: String, property_key: String },
    DropIndex { label: String, property_key: String },
}

impl std::fmt::Display for MigrationAction {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::AddLabel(l) => write!(f, "ADD LABEL {l}"),
            Self::DropLabel(l) => write!(f, "DROP LABEL {l}"),
            Self::AddEdgeType(t) => write!(f, "ADD EDGE_TYPE {t}"),
            Self::DropEdgeType(t) => write!(f, "DROP EDGE_TYPE {t}"),
            Self::AddPropertyKey(k) => write!(f, "ADD PROPERTY_KEY {k}"),
            Self::DropPropertyKey(k) => write!(f, "DROP PROPERTY_KEY {k}"),
            Self::AddIndex { label, property_key } => write!(f, "ADD INDEX ON :{label}({property_key})"),
            Self::DropIndex { label, property_key } => write!(f, "DROP INDEX ON :{label}({property_key})"),
        }
    }
}

fn parse_migration_script(content: &str) -> Result<Vec<MigrationAction>, CliError> {
    let mut actions = Vec::new();
    for (line_num, line) in content.lines().enumerate() {
        let trimmed = line.trim();
        // Skip empty lines and comments
        if trimmed.is_empty() || trimmed.starts_with('#') || trimmed.starts_with("//") {
            continue;
        }

        let upper = trimmed.to_ascii_uppercase();
        let action = if upper.starts_with("ADD LABEL ") {
            let name = trimmed[10..].trim().to_string();
            if name.is_empty() {
                return Err(CliError::Runtime(format!("line {}: ADD LABEL requires a name", line_num + 1)));
            }
            MigrationAction::AddLabel(name)
        } else if upper.starts_with("DROP LABEL ") {
            let name = trimmed[11..].trim().to_string();
            if name.is_empty() {
                return Err(CliError::Runtime(format!("line {}: DROP LABEL requires a name", line_num + 1)));
            }
            MigrationAction::DropLabel(name)
        } else if upper.starts_with("ADD EDGE_TYPE ") {
            let name = trimmed[14..].trim().to_string();
            if name.is_empty() {
                return Err(CliError::Runtime(format!("line {}: ADD EDGE_TYPE requires a name", line_num + 1)));
            }
            MigrationAction::AddEdgeType(name)
        } else if upper.starts_with("DROP EDGE_TYPE ") {
            let name = trimmed[15..].trim().to_string();
            if name.is_empty() {
                return Err(CliError::Runtime(format!("line {}: DROP EDGE_TYPE requires a name", line_num + 1)));
            }
            MigrationAction::DropEdgeType(name)
        } else if upper.starts_with("ADD PROPERTY_KEY ") {
            let name = trimmed[17..].trim().to_string();
            if name.is_empty() {
                return Err(CliError::Runtime(format!("line {}: ADD PROPERTY_KEY requires a name", line_num + 1)));
            }
            MigrationAction::AddPropertyKey(name)
        } else if upper.starts_with("DROP PROPERTY_KEY ") {
            let name = trimmed[18..].trim().to_string();
            if name.is_empty() {
                return Err(CliError::Runtime(format!("line {}: DROP PROPERTY_KEY requires a name", line_num + 1)));
            }
            MigrationAction::DropPropertyKey(name)
        } else if upper.starts_with("ADD INDEX ON ") {
            let (label, prop) = parse_index_target(&trimmed[13..])
                .ok_or_else(|| CliError::Runtime(format!(
                    "line {}: invalid ADD INDEX syntax. Expected: ADD INDEX ON :Label(property)", line_num + 1
                )))?;
            MigrationAction::AddIndex { label, property_key: prop }
        } else if upper.starts_with("DROP INDEX ON ") {
            let (label, prop) = parse_index_target(&trimmed[14..])
                .ok_or_else(|| CliError::Runtime(format!(
                    "line {}: invalid DROP INDEX syntax. Expected: DROP INDEX ON :Label(property)", line_num + 1
                )))?;
            MigrationAction::DropIndex { label, property_key: prop }
        } else {
            return Err(CliError::Runtime(format!(
                "line {}: unrecognized directive: {trimmed}", line_num + 1
            )));
        };

        actions.push(action);
    }
    Ok(actions)
}

/// Parse `:Label(property)` from an index target string.
/// Returns (label, property_key) or None if the format is invalid.
fn parse_index_target(s: &str) -> Option<(String, String)> {
    let s = s.trim();
    if !s.starts_with(':') {
        return None;
    }
    let rest = &s[1..];
    let paren_start = rest.find('(')?;
    let paren_end = rest.find(')')?;
    if paren_end <= paren_start + 1 {
        return None;
    }
    let label = rest[..paren_start].trim().to_string();
    let property = rest[paren_start + 1..paren_end].trim().to_string();
    if label.is_empty() || property.is_empty() {
        return None;
    }
    Some((label, property))
}

**Step 4: Add handle_migrate function**

```rust
fn handle_migrate(db_path: &str, script_path: &str, dry_run: bool) -> Result<String, CliError> {
    let content = std::fs::read_to_string(script_path)
        .map_err(|e| CliError::Runtime(format!("failed to read migration script '{script_path}': {e}")))?;

    let actions = parse_migration_script(&content)?;

    if actions.is_empty() {
        return Ok("migration script contains no actions".to_string());
    }

    if dry_run {
        let mut output = String::new();
        for action in &actions {
            output.push_str(&format!("[DRY-RUN] {action}\n"));
        }
        output.push_str(&format!("\n{} action(s) would be applied", actions.len()));
        return Ok(output);
    }

    // Apply mode
    if !Path::new(db_path).exists() {
        return Err(CliError::Runtime(format!(
            "error: database not found at '{db_path}'. Run 'ogdb init <path>' first."
        )));
    }
    let mut db = Database::open(db_path)
        .map_err(|e| CliError::Runtime(format!("failed to open database: {e}")))?;

    let mut applied = 0usize;
    let mut output_lines = Vec::new();

    for action in &actions {
        let result = match action {
            MigrationAction::AddLabel(name) => db.register_schema_label(name).map(|_| ()),
            MigrationAction::DropLabel(name) => db.unregister_schema_label(name).map(|_| ()),
            MigrationAction::AddEdgeType(name) => db.register_schema_edge_type(name).map(|_| ()),
            MigrationAction::DropEdgeType(name) => db.unregister_schema_edge_type(name).map(|_| ()),
            MigrationAction::AddPropertyKey(name) => db.register_schema_property_key(name).map(|_| ()),
            MigrationAction::DropPropertyKey(name) => db.unregister_schema_property_key(name).map(|_| ()),
            MigrationAction::AddIndex { label, property_key } => db.create_index(label, property_key),
            MigrationAction::DropIndex { label, property_key } => db.drop_index(label, property_key),
        };

        match result {
            Ok(()) => {
                output_lines.push(format!("[APPLIED] {action}"));
                applied += 1;
            }
            Err(e) => {
                return Err(CliError::Runtime(format!(
                    "migration failed at action {}: {action}: {e}\n{applied} action(s) applied before failure",
                    applied + 1
                )));
            }
        }
    }

    output_lines.push(format!("\n{applied} action(s) applied successfully"));
    Ok(output_lines.join("\n"))
}
```
