<objective>
Add automatic B-tree index creation to OpenGraphDB based on query frequency tracking.

Purpose: INDX-01 requires the engine to auto-create property indexes on columns that exceed a configurable query frequency threshold. Currently, indexes must be created explicitly via `CREATE INDEX` or `create_index()`. This plan adds passive tracking of property filter usage during query execution and triggers automatic index creation when the threshold is exceeded.

Output: `query_property_access_counts` field on Database, `auto_index_threshold` configuration, `maybe_auto_create_indexes()` hook called after query execution, `record_property_access()` called during physical plan building.
</objective>

```rust
    fn maybe_auto_create_indexes(&mut self) -> Result<(), DbError> {
        let threshold = match self.auto_index_threshold {
            Some(t) => t,
            None => return Ok(()),
        };

        let candidates: Vec<(String, String)> = self
            .query_property_access_counts
            .iter()
            .filter(|(_, count)| **count >= threshold)
            .map(|((label, prop), _)| (label.clone(), prop.clone()))
            .collect();

        for (label, property_key) in candidates {
            let definition = IndexDefinition {
                label: label.clone(),
                property_keys: vec![property_key.clone()],
            };
            // Skip if index already exists
            if self.meta.index_catalog.contains(&definition) {
                continue;
            }
            // Create the index
            self.create_index_definition(&label, &[property_key.clone()])?;
        }

        Ok(())
    }
```

```rust
fn collect_filtered_properties_from_plan(plan: &PhysicalPlan) -> Vec<(String, String)> {
    let mut accesses = Vec::new();
    collect_filtered_properties_recursive(plan, &mut accesses);
    accesses
}

fn collect_filtered_properties_recursive(plan: &PhysicalPlan, accesses: &mut Vec<(String, String)>) {
    match plan {
        PhysicalPlan::PhysicalFilter { input, predicate, .. } => {
            // Check if input is a scan with a label
            if let PhysicalPlan::PhysicalScan { label, variable, .. } = input.as_ref() {
                // Extract property keys referenced in the predicate
                let mut predicates = Vec::new();
                collect_index_predicates_for_variable(predicate, variable, &mut predicates);
                for pred in &predicates {
                    accesses.push((label.clone(), pred.property_key.clone()));
                }
            }
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalScan { .. } => {}
        PhysicalPlan::PhysicalExpand { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalProjection { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalLimit { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalSkip { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalSort { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalDistinct { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalAggregate { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalUnwind { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalWith { input, .. } => {
            collect_filtered_properties_recursive(input, accesses);
        }
        PhysicalPlan::PhysicalHashJoin { left, right, .. } => {
            collect_filtered_properties_recursive(left, accesses);
            collect_filtered_properties_recursive(right, accesses);
        }
        PhysicalPlan::PhysicalCartesianProduct { left, right, .. } => {
            collect_filtered_properties_recursive(left, accesses);
            collect_filtered_properties_recursive(right, accesses);
        }
        PhysicalPlan::PhysicalOptionalMatch { input, optional, .. } => {
            collect_filtered_properties_recursive(input, accesses);
            collect_filtered_properties_recursive(optional, accesses);
        }
        PhysicalPlan::PhysicalUnion { branches, .. } => {
            for branch in branches {
                collect_filtered_properties_recursive(branch, accesses);
            }
        }
        _ => {
            // For any other variant, just return — we only track Filter+Scan combos
        }
    }
}
```

```rust
#[test]
fn auto_index_creates_index_after_threshold_queries() {
    let dir = tempdir().unwrap();
    let path = dir.path().join("autoindex.ogdb");
    let mut db = Database::init(&path, Header::default_v1()).unwrap();

    // Set a low threshold for testing
    db.set_auto_index_threshold(Some(5));

    // Create some data
    {
        let mut tx = db.begin_write();
        for i in 0..10 {
            let node_id = tx.create_node().unwrap();
            tx.set_node_labels(node_id, BTreeSet::from(["Person".to_string()])).unwrap();
            tx.set_node_properties(node_id, BTreeMap::from([
                ("name".to_string(), PropertyValue::String(format!("Person{i}"))),
            ])).unwrap();
        }
        tx.commit().unwrap();
    }

    // Verify no index exists yet
    assert!(db.list_indexes().is_empty());

    // Run filter queries below threshold
    for _ in 0..4 {
        let _ = db.query("MATCH (p:Person) WHERE p.name = 'Person0' RETURN p");
    }
    assert!(db.list_indexes().is_empty(), "no index before threshold");

    // Run one more query to hit threshold
    let _ = db.query("MATCH (p:Person) WHERE p.name = 'Person0' RETURN p");

    // Now an index should have been auto-created
    let indexes = db.list_indexes();
    assert_eq!(indexes.len(), 1);
    assert_eq!(indexes[0].label, "Person");
    assert_eq!(indexes[0].property_keys, vec!["name".to_string()]);
}
```

```rust
fn auto_index_tracks_multiple_properties_independently() {
    let dir = tempdir().unwrap();
    let path = dir.path().join("autoindex_multi.ogdb");
    let mut db = Database::init(&path, Header::default_v1()).unwrap();

        let node_id = tx.create_node().unwrap();
        tx.set_node_labels(node_id, BTreeSet::from(["Person".to_string()])).unwrap();

            ("age".to_string(), PropertyValue::I64(30)),
        ])).unwrap();
        tx.commit().unwrap();
    }

    // Query name 3 times (should trigger)
    for _ in 0..3 {
        let _ = db.query("MATCH (p:Person) WHERE p.name = 'Alice' RETURN p");
    }
    // Query age 2 times (should NOT trigger)
    for _ in 0..2 {
        let _ = db.query("MATCH (p:Person) WHERE p.age = 30 RETURN p");

    assert_eq!(indexes.len(), 1, "only name index created, not age");
    assert_eq!(indexes[0].property_keys, vec!["name".to_string()]);

    // One more age query to trigger
    let _ = db.query("MATCH (p:Person) WHERE p.age = 30 RETURN p");
    let indexes = db.list_indexes();
    assert_eq!(indexes.len(), 2, "both name and age indexes now exist");
}
```
