       - Keep all existing types unchanged.

         ```typescript
         export function useSchemaQuery() {
           const client = useApiClient()
           const serverUrl = useSettingsStore((s) => s.serverUrl)
           return useQuery({
             queryKey: ['schema', serverUrl],
             queryFn: () => client.schema() as Promise<SchemaResponse>,
             staleTime: 60_000,
             retry: 1,
             enabled: true,
           })
         }
         ```

         - Get `schema` from `useSchemaAsDbSchema()`

         - `handleExecute` callback: receives `cmd: string` from CypherEditor's `onExecute` prop:
           1. Call `prepareCypherQuery(cmd, resultLimit)` to get `finalQuery`
           2. If `!finalQuery` return early
           3. Call `addToHistory(cmd.trim())` (store the original query without LIMIT appended)
           4. Call `onRunQuery(finalQuery)`

         - Render:
           ```tsx
           <div className="border-b bg-card p-3">
             <div className="flex flex-col gap-2 sm:flex-row">
               <div className="flex-1 overflow-hidden rounded-md border border-input">
                 <CypherEditor
                   value={currentQuery}
                   onChange={(val) => setCurrentQuery(val)}
                   onExecute={handleExecute}
                   history={history}
                   schema={schema}
                   theme={resolvedTheme === 'dark' ? 'dark' : 'light'}
                   lint={true}
                   placeholder="Enter a Cypher query... e.g., MATCH (n) RETURN n LIMIT 25"
                   ariaLabel="Cypher query editor"
                   className="min-h-[80px] max-h-[200px]"
                 />
               </div>
               <Button
                 onClick={() => handleExecute(currentQuery)}
                 disabled={isRunning || !currentQuery.trim()}
                 className="w-full sm:w-auto sm:self-end"
               >
                 {isRunning ? (
                   <Loader2 className="h-4 w-4 animate-spin" />
                 ) : (
                   <Play className="h-4 w-4" />
                 )}
                 <span className="ml-1.5">Run</span>
               </Button>
             </div>
           </div>
           ```
