**Step 1: Create `skills/evals/runner.ts`**

The eval runner loads YAML test cases and evaluates LLM outputs against expected criteria.

Since this is a skill eval framework (not a full LLM test harness), the runner does:
1. Loads eval YAML file
2. For each test case, constructs the prompt (with and without skill content)
3. Outputs the constructed prompts for manual or automated LLM evaluation
4. Scores responses against expected patterns (regex, keyword presence)
5. Produces a summary report

```typescript
import { readFileSync, existsSync } from "node:fs";
import { join, dirname, resolve } from "node:path";
import { fileURLToPath } from "node:url";

const __dirname = dirname(fileURLToPath(import.meta.url));

export interface EvalCase {
  name: string;
  difficulty: string;
  input: string;
  context?: {
    schema?: {
      labels: string[];
      edge_types: string[];
      property_keys: string[];
    };
    data?: string;
  };
  expected: {
    must_contain?: string[];
    must_not_contain?: string[];
    pattern?: string;
  };
  scoring: Record<string, number>;
}

export interface EvalSuite {
  skill: string;
  version: string;
  description: string;
  cases: EvalCase[];
}

export interface EvalResult {
  caseName: string;
  passed: boolean;
  score: number;
  maxScore: number;
  details: string[];
}

// Parse YAML manually (no yaml dependency needed for simple structure)
// This uses a minimal YAML parser for the eval format
export function parseEvalYaml(content: string): EvalSuite {
  // For the eval framework, we use JSON-compatible YAML subset
  // The actual YAML parsing will use a simple line-based approach
  // For production, consider adding js-yaml as a dependency

  // Simple approach: eval files can also be written as JSON
  // The YAML format is for human readability; runner accepts both
  try {
    return JSON.parse(content) as EvalSuite;
  } catch {
    // Fallback: minimal YAML parsing for the eval format
    throw new Error("Eval files must be valid JSON or use the eval runner with a YAML parser. Install js-yaml for full YAML support.");
  }
}

export function scoreResponse(response: string, evalCase: EvalCase): EvalResult {
  const details: string[] = [];
  let score = 0;
  let maxScore = 0;

  const expected = evalCase.expected;

  // Check must_contain
  if (expected.must_contain) {
    for (const keyword of expected.must_contain) {
      maxScore += 1;
      if (response.includes(keyword)) {
        score += 1;
        details.push(`PASS: contains "${keyword}"`);
      } else {
        details.push(`FAIL: missing "${keyword}"`);
      }
    }
  }

  // Check must_not_contain
  if (expected.must_not_contain) {
    for (const keyword of expected.must_not_contain) {
      maxScore += 1;
      if (!response.includes(keyword)) {
        score += 1;
        details.push(`PASS: does not contain "${keyword}"`);
      } else {
        details.push(`FAIL: should not contain "${keyword}"`);
      }
    }
  }

  // Check pattern
  if (expected.pattern) {
    maxScore += 2; // patterns are worth more
    try {
      const regex = new RegExp(expected.pattern, "i");
      if (regex.test(response)) {
        score += 2;
        details.push(`PASS: matches pattern /${expected.pattern}/`);
      } else {
        details.push(`FAIL: does not match pattern /${expected.pattern}/`);
      }
    } catch {
      details.push(`SKIP: invalid pattern /${expected.pattern}/`);
    }
  }

  // Add scoring weights
  for (const [criterion, weight] of Object.entries(evalCase.scoring)) {
    maxScore += weight;
    // Scoring weights are evaluated by the LLM judge, not pattern matching
    // They represent criteria for A/B comparison
    details.push(`MANUAL: "${criterion}" (weight: ${weight}) requires LLM judge`);
  }

  return {
    caseName: evalCase.name,
    passed: score >= maxScore * 0.6, // 60% threshold
    score,
    maxScore,
    details,
  };
}

export function generatePrompt(
  evalCase: EvalCase,
  skillContent: string | null,
  skillName: string
): string {
  const parts: string[] = [];

  if (skillContent) {
    parts.push(`<skill>\n${skillContent}\n</skill>\n`);
  }

  if (evalCase.context?.schema) {
    const schema = evalCase.context.schema;
    parts.push(`<database_schema>`);
    parts.push(`Labels: ${schema.labels.join(", ")}`);
    parts.push(`Relationship Types: ${schema.edge_types.join(", ")}`);
    parts.push(`Property Keys: ${schema.property_keys.join(", ")}`);
    parts.push(`</database_schema>\n`);
  }

  if (evalCase.context?.data) {
    parts.push(`<data>\n${evalCase.context.data}\n</data>\n`);
  }

  parts.push(`User: ${evalCase.input}`);

  return parts.join("\n");
}

export function printReport(
  skillName: string,
  results: EvalResult[],
  mode: "with-skill" | "without-skill" | "comparison"
): void {
  console.log(`\n${"=".repeat(60)}`);
  console.log(`Eval Report: ${skillName} (${mode})`);
  console.log(`${"=".repeat(60)}\n`);

  let totalPassed = 0;
  let totalCases = results.length;

  for (const result of results) {
    const status = result.passed ? "PASS" : "FAIL";
    console.log(`  [${status}] ${result.caseName} (${result.score}/${result.maxScore})`);
    for (const detail of result.details) {
      console.log(`         ${detail}`);
    }
  }

  totalPassed = results.filter((r) => r.passed).length;

  console.log(`\nSummary: ${totalPassed}/${totalCases} passed (${Math.round((totalPassed / totalCases) * 100)}%)\n`);
}
```

**Step 2: Create `skills/src/eval.ts`**

CLI command that runs evals:

```typescript
import { readFileSync, readdirSync, existsSync } from "node:fs";
import { join, dirname, resolve } from "node:path";
import { fileURLToPath } from "node:url";
import { parseEvalYaml, generatePrompt, printReport } from "../evals/runner.js";

const __dirname = dirname(fileURLToPath(import.meta.url));
const SKILLS_ROOT = resolve(__dirname, "..");
const EVALS_DIR = join(SKILLS_ROOT, "evals");

interface EvalOptions {
  skillName?: string;
  mode?: "prompts" | "score";
  responseFile?: string;
}

function loadSkillContent(skillName: string): string {
  const skillDir = join(SKILLS_ROOT, skillName);
  const skillMd = readFileSync(join(skillDir, "SKILL.md"), "utf-8");
  const rulesDir = join(skillDir, "rules");
  let combined = skillMd;

  if (existsSync(rulesDir)) {
    const ruleFiles = readdirSync(rulesDir).filter((f) => f.endsWith(".md")).sort();
    for (const ruleFile of ruleFiles) {
      combined += "\n\n---\n\n" + readFileSync(join(rulesDir, ruleFile), "utf-8");
    }
  }

  return combined;
}

export async function runEvals(options: EvalOptions): Promise<void> {
  const evalFiles = readdirSync(EVALS_DIR)
    .filter((f) => f.endsWith(".eval.yaml") || f.endsWith(".eval.json"))
    .filter((f) => !options.skillName || f.startsWith(options.skillName));

  if (evalFiles.length === 0) {
    console.log("No eval files found.");
    return;
  }

  for (const evalFile of evalFiles) {
    const content = readFileSync(join(EVALS_DIR, evalFile), "utf-8");
    const suite = parseEvalYaml(content);

    console.log(`\nRunning evals for: ${suite.skill} (${suite.cases.length} cases)`);

    if (options.mode === "prompts") {
      // Generate prompts mode: output prompts for manual LLM testing
      console.log("\n--- Prompts (WITH skill) ---\n");
      const skillContent = loadSkillContent(suite.skill);
      for (const evalCase of suite.cases) {
        console.log(`\n### ${evalCase.name} (${evalCase.difficulty}) ###\n`);
        console.log(generatePrompt(evalCase, skillContent, suite.skill));
        console.log("\n---\n");
      }

      console.log("\n--- Prompts (WITHOUT skill) ---\n");
      for (const evalCase of suite.cases) {
        console.log(`\n### ${evalCase.name} (${evalCase.difficulty}) ###\n`);
        console.log(generatePrompt(evalCase, null, suite.skill));
        console.log("\n---\n");
      }
    } else {
      // Score mode: score responses from a file
      if (options.responseFile) {
        // Score pre-recorded responses (future: automated LLM evaluation)
        console.log(`Scoring responses from: ${options.responseFile}`);
        console.log("(Response scoring not yet implemented. Use 'prompts' mode to generate test prompts.)");
      } else {
        console.log("Usage:");
        console.log("  npx @opengraphdb/skills eval prompts [skill-name]  # Generate test prompts");
        console.log("  npx @opengraphdb/skills eval score [responses.json]  # Score responses");
      }
    }
  }
}
```

**IMPORTANT design notes:**
- Eval files use YAML for human readability but the runner also accepts JSON. For v0.1.0, convert the YAML eval files to JSON format to avoid adding a YAML parsing dependency. Use JSON with comments-friendly formatting.
