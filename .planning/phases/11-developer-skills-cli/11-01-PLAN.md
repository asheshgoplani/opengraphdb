**Step 3: Create `skills/src/install.ts`**

The install script reads skill directories (SKILL.md + rules/*.md), detects the AI platform, and writes the appropriate configuration files.

```typescript
import { readFileSync, writeFileSync, mkdirSync, existsSync, readdirSync } from "node:fs";
import { join, dirname, resolve } from "node:path";
import { fileURLToPath } from "node:url";

const __dirname = dirname(fileURLToPath(import.meta.url));
// Skills live alongside dist/ in the package root
const SKILLS_ROOT = resolve(__dirname, "..");

const ALL_SKILLS = ["ogdb-cypher", "graph-explore", "schema-advisor", "data-import"];

type Platform = "claude" | "cursor" | "copilot" | "codex";

interface InstallOptions {
  platform?: string;
  skillNames?: string[];
}

function detectPlatform(): Platform {
  // Check for platform indicators in the current project
  if (existsSync(".claude") || existsSync("CLAUDE.md")) return "claude";
  if (existsSync(".cursor") || existsSync(".cursorrules")) return "cursor";
  if (existsSync(".github/copilot-instructions.md")) return "copilot";
  if (existsSync(".codex")) return "codex";
  // Default to Claude Code format (most structured)
  return "claude";
}

function loadSkill(skillName: string): string {
  const skillDir = join(SKILLS_ROOT, skillName);
  if (!existsSync(skillDir)) {
    throw new Error(`Skill not found: ${skillName}. Run 'npx @opengraphdb/skills list' to see available skills.`);
  }

  const skillMd = readFileSync(join(skillDir, "SKILL.md"), "utf-8");
  const rulesDir = join(skillDir, "rules");
  let combined = skillMd;

  if (existsSync(rulesDir)) {
    const ruleFiles = readdirSync(rulesDir)
      .filter((f) => f.endsWith(".md"))
      .sort();

    for (const ruleFile of ruleFiles) {
      const ruleContent = readFileSync(join(rulesDir, ruleFile), "utf-8");
      combined += `\n\n---\n\n${ruleContent}`;
    }
  }

  return combined;
}

function installClaude(skills: string[]): void {
  for (const skillName of skills) {
    const skillDir = join(SKILLS_ROOT, skillName);
    const targetDir = join(".claude", "skills", skillName);
    mkdirSync(targetDir, { recursive: true });

    // Copy SKILL.md
    const skillMd = readFileSync(join(skillDir, "SKILL.md"), "utf-8");
    writeFileSync(join(targetDir, "SKILL.md"), skillMd);

    // Copy rules/
    const rulesDir = join(skillDir, "rules");
    if (existsSync(rulesDir)) {
      const targetRulesDir = join(targetDir, "rules");
      mkdirSync(targetRulesDir, { recursive: true });
      for (const ruleFile of readdirSync(rulesDir).filter((f) => f.endsWith(".md"))) {
        const content = readFileSync(join(rulesDir, ruleFile), "utf-8");
        writeFileSync(join(targetRulesDir, ruleFile), content);
      }
    }

    console.log(`  Installed ${skillName} → .claude/skills/${skillName}/`);
  }
}

function installSingleFile(skills: string[], targetPath: string, header: string): void {
  const sections: string[] = [header];

  for (const skillName of skills) {
    const content = loadSkill(skillName);
    sections.push(`\n## Skill: ${skillName}\n\n${content}`);
  }

  const targetDir = dirname(targetPath);
  if (targetDir !== ".") {
    mkdirSync(targetDir, { recursive: true });
  }
  writeFileSync(targetPath, sections.join("\n"));
  console.log(`  Installed ${skills.length} skill(s) → ${targetPath}`);
}

export async function install(options: InstallOptions): Promise<void> {
  const platform = (options.platform as Platform) || detectPlatform();
  const skills = options.skillNames?.length
    ? options.skillNames.filter((s) => ALL_SKILLS.includes(s))
    : ALL_SKILLS.filter((s) => existsSync(join(SKILLS_ROOT, s)));

  if (skills.length === 0) {
    console.log("No skills to install. Run 'npx @opengraphdb/skills list' to see available skills.");
    return;
  }

  console.log(`\nInstalling ${skills.length} OpenGraphDB skill(s) for ${platform}...\n`);

  switch (platform) {
    case "claude":
      installClaude(skills);
      break;
    case "cursor":
      installSingleFile(
        skills,
        ".cursorrules",
        "# OpenGraphDB Skills\n\nThese rules teach you how to work with OpenGraphDB, a graph database with Cypher query support.\n"
      );
      break;
    case "copilot":
      installSingleFile(
        skills,
        ".github/copilot-instructions.md",
        "# OpenGraphDB Skills\n\nThese instructions teach you how to work with OpenGraphDB, a graph database with Cypher query support.\n"
      );
      break;
    case "codex":
      installSingleFile(
        skills,
        ".codex/instructions.md",
        "# OpenGraphDB Skills\n\nThese instructions teach you how to work with OpenGraphDB, a graph database with Cypher query support.\n"
      );
      break;
    default:
      throw new Error(`Unknown platform: ${platform}. Use: claude, cursor, copilot, or codex`);
  }

  console.log("\nDone! Your AI coding tool now has OpenGraphDB expertise.");
  console.log("Try asking: \"Show me all node labels in the database\"");
}
```

**Step 2: Create `skills/ogdb-cypher/rules/cypher-patterns.md`**

Detailed Cypher query patterns for OpenGraphDB. Include:

1. **Node Patterns**: `(n)`, `(n:Label)`, `(n:Label {prop: value})`, `(n:Label {prop: $param})`
2. **Relationship Patterns**: `-[r]->`, `-[r:TYPE]->`, `-[r:TYPE*1..3]->`, `<-[r]-`, `-[r]-`
3. **MATCH + WHERE Patterns**:
   - Property equality: `WHERE n.name = 'Alice'`
   - String matching: `WHERE n.name CONTAINS 'ali'`, `STARTS WITH`, `ENDS WITH`
   - Comparison operators: `>`, `<`, `>=`, `<=`, `<>`
   - Boolean: `AND`, `OR`, `NOT`, `XOR`
   - NULL checks: `IS NULL`, `IS NOT NULL`
   - IN lists: `WHERE n.age IN [25, 30, 35]`
   - Regular expressions: `WHERE n.name =~ '(?i)tom.*'`
4. **CREATE Patterns**: Creating nodes, relationships, setting properties inline
5. **MERGE Patterns**: MERGE with ON CREATE SET and ON MATCH SET
6. **SET/REMOVE Patterns**: Property updates, label operations
7. **DELETE Patterns**: DELETE vs DETACH DELETE
8. **Aggregation Patterns**: GROUP BY (implicit), count, sum, avg, min, max, collect
9. **WITH Chaining**: Using WITH for multi-step queries, aliasing, filtering between clauses
10. **UNWIND Patterns**: List processing, parameter expansion
11. **Path Patterns**: Variable-length paths `*1..3`, shortest path queries
12. **OpenGraphDB Extensions**:
    - Temporal queries: `MATCH (n:Person) AT TIME 1700000000 RETURN n`
    - Vector search: via `vector_search` MCP tool (not inline Cypher)
    - Text search: via `text_search` MCP tool (not inline Cypher)
    - RDF import: via `import_rdf` MCP tool
    - Index creation: `CREATE INDEX ON :Label(property)`

Target: ~150-200 lines. Be specific with examples, not vague.

**Step 3: Create `skills/ogdb-cypher/rules/query-optimization.md`**

Performance guidance for OpenGraphDB:

1. **Always use LIMIT**: Prevent returning unbounded results. Default to LIMIT 25 for exploration, LIMIT 100 for analysis.
2. **Index-aware queries**: Use labeled node patterns `(n:Label)` not bare `(n)`. OpenGraphDB uses label bitmaps for fast filtering.
3. **Property filtering order**: Put indexed property predicates first in WHERE clauses.
4. **Avoid `MATCH (n) RETURN n`**: Always include a label filter or property filter.
5. **Variable-length paths**: Use bounded ranges `*1..3` not unbounded `*`. Max recommended: 5 hops.
6. **Aggregation before expansion**: Use WITH to aggregate before joining with more patterns.
7. **OPTIONAL MATCH vs MATCH**: Use OPTIONAL MATCH for nullable relationships to avoid filtering out nodes.
8. **COUNT optimization**: `RETURN count(n)` is faster than `RETURN n` followed by client-side counting.
9. **Property access**: Access only needed properties (`RETURN n.name, n.age`) rather than returning the full node (`RETURN n`).
10. **Batch operations**: Use UNWIND for batch inserts rather than multiple CREATE statements.

Target: ~80-100 lines.

**Step 4: Create `skills/ogdb-cypher/rules/error-prevention.md`**

Common mistakes and how to avoid them:

1. **Missing labels in MATCH**: `MATCH (n) RETURN n` scans all nodes. Always specify a label.
2. **Unbounded variable-length paths**: `MATCH (a)-[*]->(b)` can explode. Always bound: `[*1..3]`.
3. **DELETE without DETACH**: `DELETE n` fails if node has relationships. Use `DETACH DELETE n`.
4. **Property type mismatches**: String vs integer comparison (`WHERE n.age = '30'` vs `WHERE n.age = 30`).
5. **Case sensitivity**: Cypher keywords are case-insensitive but labels and property keys are case-sensitive. `:Person` is not `:person`.
6. **Missing RETURN clause**: Every read query needs RETURN. Write queries (CREATE/DELETE/SET) return nothing unless you add RETURN.
7. **Overwriting properties with SET**: `SET n = {name: 'Bob'}` replaces ALL properties. Use `SET n.name = 'Bob'` for single property updates.
8. **MERGE on too many properties**: `MERGE (n:Person {name: 'Alice', age: 30, city: 'NYC'})` creates duplicates if any property differs. MERGE on unique identifiers only.
9. **toString() for CONTAINS**: OpenGraphDB CONTAINS works on strings. If a property might be non-string, wrap in `toString()`.
10. **Semicolons between statements**: OpenGraphDB supports multi-statement queries separated by `;`. Each statement executes independently.

Target: ~60-80 lines.
