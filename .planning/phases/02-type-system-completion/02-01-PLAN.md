/// Parses "YYYY-MM-DD" -> days since 1970-01-01
fn parse_date_literal(s: &str) -> Option<i32> {
    let bytes = s.as_bytes();
    if bytes.len() < 10 || bytes[4] != b'-' || bytes[7] != b'-' { return None; }
    let year: i32 = s[0..4].parse().ok()?;
    let month: u32 = s[5..7].parse().ok()?;
    let day: u32 = s[8..10].parse().ok()?;
    if month < 1 || month > 12 || day < 1 || day > 31 { return None; }
    Some(date_to_days_since_epoch(year, month, day))
}

/// Parses "YYYY-MM-DDTHH:MM:SSZ" or "YYYY-MM-DDTHH:MM:SS+HH:MM" -> (micros_since_epoch, tz_offset_minutes)
fn parse_datetime_literal(s: &str) -> Option<(i64, i16)> {
    // Find 'T' separator
    let t_pos = s.find('T').or_else(|| s.find('t'))?;
    let date_part = &s[..t_pos];
    let time_and_tz = &s[t_pos + 1..];

    // Parse date
    let days = parse_date_literal(date_part)?;

    // Find timezone suffix: Z, +HH:MM, -HH:MM
    let (time_part, tz_offset_minutes) = if time_and_tz.ends_with('Z') || time_and_tz.ends_with('z') {
        (&time_and_tz[..time_and_tz.len() - 1], 0i16)
    } else if let Some(plus_pos) = time_and_tz.rfind('+') {
        let tz_str = &time_and_tz[plus_pos + 1..];
        let tz_mins = parse_tz_offset(tz_str)?;
        (&time_and_tz[..plus_pos], tz_mins)
    } else if let Some(minus_pos) = time_and_tz.rfind('-') {
        // Make sure this minus is in the timezone, not in the time (HH:MM:SS has no minus)
        if minus_pos > 2 { // After at least HH:M
            let tz_str = &time_and_tz[minus_pos + 1..];
            let tz_mins = parse_tz_offset(tz_str)?;
            (&time_and_tz[..minus_pos], -tz_mins)
        } else {
            return None;
        }
    } else {
        // No timezone = assume UTC
        (time_and_tz, 0i16)
    };

    // Parse time: HH:MM:SS or HH:MM:SS.fractional
    let time_parts: Vec<&str> = time_part.split(':').collect();
    if time_parts.len() < 2 { return None; }
    let hours: i64 = time_parts[0].parse().ok()?;
    let minutes: i64 = time_parts[1].parse().ok()?;
    let (seconds, frac_micros) = if time_parts.len() >= 3 {
        let sec_str = time_parts[2];
        if let Some(dot_pos) = sec_str.find('.') {
            let secs: i64 = sec_str[..dot_pos].parse().ok()?;
            let frac_str = &sec_str[dot_pos + 1..];
            // Pad or truncate to 6 digits for microseconds
            let padded = format!("{:0<6}", frac_str);
            let micros: i64 = padded[..6].parse().ok()?;
            (secs, micros)
        } else {
            (sec_str.parse().ok()?, 0i64)
        }
    } else {
        (0i64, 0i64)
    };

    let day_micros = ((hours * 3600 + minutes * 60 + seconds) * 1_000_000) + frac_micros;
    let total_micros = (days as i64) * 86_400_000_000i64 + day_micros;
    // Adjust for timezone: stored as UTC
    let utc_micros = total_micros - (tz_offset_minutes as i64) * 60_000_000i64;

    Some((utc_micros, tz_offset_minutes))
}

/// Parses "HH:MM" or "HHMM" timezone offset string -> minutes
fn parse_tz_offset(s: &str) -> Option<i16> {
    if s.len() == 5 && s.as_bytes()[2] == b':' {
        let h: i16 = s[..2].parse().ok()?;
        let m: i16 = s[3..5].parse().ok()?;
        Some(h * 60 + m)
    } else if s.len() == 4 {
        let h: i16 = s[..2].parse().ok()?;
        let m: i16 = s[2..4].parse().ok()?;
        Some(h * 60 + m)
    } else {
        None
    }
}

/// Converts days-since-epoch to "YYYY-MM-DD" string
fn days_to_date_string(days: i32) -> String {
    // Civil date from days since 1970-01-01 using the algorithm from Howard Hinnant
    let z = days as i64 + 719468; // days from 0000-03-01
    let era = (if z >= 0 { z } else { z - 146096 }) / 146097;
    let doe = (z - era * 146097) as u32; // day of era [0, 146096]
    let yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
    let y = yoe as i64 + era * 400;
    let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
    let mp = (5 * doy + 2) / 153;
    let d = doy - (153 * mp + 2) / 5 + 1;
    let m = if mp < 10 { mp + 3 } else { mp - 9 };
    let y = if m <= 2 { y + 1 } else { y };
    format!("{:04}-{:02}-{:02}", y, m, d)
}

/// Converts microseconds-since-epoch + tz offset to ISO 8601 datetime string
fn micros_to_datetime_string(micros: i64, tz_offset_minutes: i16) -> String {
    // Convert UTC micros to local time by adding tz offset
    let local_micros = micros + (tz_offset_minutes as i64) * 60_000_000i64;
    let total_secs = local_micros.div_euclid(1_000_000);
    let frac_micros = local_micros.rem_euclid(1_000_000);
    let days = total_secs.div_euclid(86400) as i32;
    let day_secs = total_secs.rem_euclid(86400);
    let hours = day_secs / 3600;
    let minutes = (day_secs % 3600) / 60;
    let seconds = day_secs % 60;
    let date_str = days_to_date_string(days);

    let tz_str = if tz_offset_minutes == 0 {
        "Z".to_string()
    } else {
        let sign = if tz_offset_minutes > 0 { '+' } else { '-' };
        let abs_mins = tz_offset_minutes.unsigned_abs();
        format!("{}{:02}:{:02}", sign, abs_mins / 60, abs_mins % 60)
    };

    if frac_micros == 0 {
        format!("{date_str}T{hours:02}:{minutes:02}:{seconds:02}{tz_str}")
    } else {
        format!("{date_str}T{hours:02}:{minutes:02}:{seconds:02}.{frac_micros:06}{tz_str}")
    }
}

/// Converts a civil date to days since 1970-01-01
fn date_to_days_since_epoch(year: i32, month: u32, day: u32) -> i32 {
    // Howard Hinnant's algorithm (civil_from_days inverse)
    let y = if month <= 2 { year as i64 - 1 } else { year as i64 };
    let era = (if y >= 0 { y } else { y - 399 }) / 400;
    let yoe = (y - era * 400) as u32;
    let m = month;
    let doy = (153 * (if m > 2 { m - 3 } else { m + 9 }) + 2) / 5 + day - 1;
    let doe = yoe * 365 + yoe / 4 - yoe / 100 + doy;
    let days = era * 146097 + doe as i64 - 719468;
    days as i32
}
