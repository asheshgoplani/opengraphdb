```rust
/// Parses ISO 8601 duration "P[n]Y[n]M[n]DT[n]H[n]M[n]S" -> (months, days, nanos)
fn parse_duration_literal(s: &str) -> Option<(i64, i64, i64)> {
    let s = s.trim();
    if !s.starts_with('P') && !s.starts_with('p') { return None; }
    let s = &s[1..]; // skip 'P'

    let mut months: i64 = 0;
    let mut days: i64 = 0;
    let mut nanos: i64 = 0;
    let mut in_time = false;
    let mut num_buf = String::new();

    for ch in s.chars() {
        match ch {
            'T' | 't' => {
                in_time = true;
            }
            '0'..='9' | '.' => {
                num_buf.push(ch);
            }
            'Y' | 'y' if !in_time => {
                let years: i64 = num_buf.parse().ok()?;
                months += years * 12;
                num_buf.clear();
            }
            'M' | 'm' if !in_time => {
                let m: i64 = num_buf.parse().ok()?;
                months += m;
                num_buf.clear();
            }
            'W' | 'w' if !in_time => {
                let weeks: i64 = num_buf.parse().ok()?;
                days += weeks * 7;
                num_buf.clear();
            }
            'D' | 'd' if !in_time => {
                let d: i64 = num_buf.parse().ok()?;
                days += d;
                num_buf.clear();
            }
            'H' | 'h' if in_time => {
                let h: f64 = num_buf.parse().ok()?;
                nanos += (h * 3_600_000_000_000.0) as i64;
                num_buf.clear();
            }
            'M' | 'm' if in_time => {
                let m: f64 = num_buf.parse().ok()?;
                nanos += (m * 60_000_000_000.0) as i64;
                num_buf.clear();
            }
            'S' | 's' if in_time => {
                let secs: f64 = num_buf.parse().ok()?;
                nanos += (secs * 1_000_000_000.0) as i64;
                num_buf.clear();
            }
            _ => return None, // unexpected character
        }
    }

    // If there's leftover in num_buf, the format is invalid
    if !num_buf.is_empty() { return None; }

    Some((months, days, nanos))
}

/// Formats duration components as ISO 8601 string
fn format_duration(months: i64, days: i64, nanos: i64) -> String {
    let mut result = String::from("P");
    let years = months / 12;
    let remaining_months = months % 12;
    if years != 0 { result.push_str(&format!("{years}Y")); }
    if remaining_months != 0 { result.push_str(&format!("{remaining_months}M")); }
    if days != 0 { result.push_str(&format!("{days}D")); }

    if nanos != 0 {
        result.push('T');
        let total_secs = nanos / 1_000_000_000;
        let remaining_nanos = nanos % 1_000_000_000;
        let hours = total_secs / 3600;
        let mins = (total_secs % 3600) / 60;
        let secs = total_secs % 60;
        if hours != 0 { result.push_str(&format!("{hours}H")); }
        if mins != 0 { result.push_str(&format!("{mins}M")); }
        if secs != 0 || remaining_nanos != 0 {
            if remaining_nanos != 0 {
                let frac = format!("{:09}", remaining_nanos).trim_end_matches('0').to_string();
                result.push_str(&format!("{secs}.{frac}S"));
            } else {
                result.push_str(&format!("{secs}S"));
            }
        }
    }

    // Edge case: zero duration
    if result == "P" { result.push_str("T0S"); }

    result
}
```
