<task type="auto">
  <name>Task 1: Install Playwright and create configuration</name>
  <files>frontend/package.json, frontend/playwright.config.ts</files>
  <action>
1. Install Playwright:
   ```bash
   cd frontend && npm install -D @playwright/test
   ```

2. Install browsers:
   ```bash
   cd frontend && npx playwright install chromium
   ```

3. Create `frontend/playwright.config.ts`:
   ```typescript
   import { defineConfig, devices } from '@playwright/test'

   export default defineConfig({
     testDir: './e2e',
     fullyParallel: true,
     forbidOnly: !!process.env.CI,
     retries: process.env.CI ? 1 : 0,
     workers: 1,
     reporter: 'list',
     use: {
       baseURL: 'http://localhost:5173',
       screenshot: 'only-on-failure',
       trace: 'on-first-retry',
       viewport: { width: 1280, height: 800 },
     },
     projects: [
       {
         name: 'chromium',
         use: { ...devices['Desktop Chrome'] },
       },
     ],
     webServer: {
       command: 'npm run dev',
       url: 'http://localhost:5173',
       reuseExistingServer: !process.env.CI,
       timeout: 30000,
     },
   })
   ```

4. Add script to package.json:
   ```json
   "test:e2e": "playwright test"
   ```

5. Create `frontend/e2e/` directory and `frontend/e2e/screenshots/` directory (add `.gitkeep` to screenshots).
  </action>

    <automated>cd frontend && npx playwright --version 2>&1</automated>

  <action>
1. Create `frontend/e2e/landing.spec.ts`:
   ```typescript
   import { test, expect } from '@playwright/test'

   test.describe('Landing Page', () => {
     test.beforeEach(async ({ page }) => {
       await page.goto('/')
       // Wait for page to fully render
       await page.waitForLoadState('networkidle')
     })

     test('renders hero section with title', async ({ page }) => {
       const hero = page.locator('h1')
       await expect(hero).toBeVisible()
       await expect(hero).toContainText('Graph Database')
       await page.screenshot({ path: 'e2e/screenshots/landing-hero-light.png', fullPage: false })
     })

     test('renders showcase section with 3 dataset cards', async ({ page }) => {
       const showcase = page.locator('#use-cases')
       await showcase.scrollIntoViewIfNeeded()
       await page.waitForTimeout(600) // wait for scroll animations
       const cards = showcase.locator('[data-testid="showcase-card"]')
       await expect(cards).toHaveCount(3)
       await page.screenshot({ path: 'e2e/screenshots/landing-showcase-light.png', fullPage: false })
     })

     test('renders features section with 4 feature cards', async ({ page }) => {
       const features = page.locator('#features')
       await features.scrollIntoViewIfNeeded()
       await page.waitForTimeout(600)
       const cards = features.locator('.card, [class*="Card"]').or(features.locator('[data-testid="feature-card"]'))
       await expect(cards).toHaveCount(4)
       await page.screenshot({ path: 'e2e/screenshots/landing-features-light.png', fullPage: false })
     })

     test('renders getting started section', async ({ page }) => {
       const gettingStarted = page.locator('#get-started')
       await gettingStarted.scrollIntoViewIfNeeded()
       await page.waitForTimeout(600)
       await expect(gettingStarted).toBeVisible()
       await page.screenshot({ path: 'e2e/screenshots/landing-getting-started-light.png', fullPage: false })
     })

     test('full page screenshot - light mode', async ({ page }) => {
       await page.screenshot({ path: 'e2e/screenshots/landing-full-light.png', fullPage: true })
     })

     test('full page screenshot - dark mode', async ({ page }) => {
       // Toggle to dark mode via DOM class
       await page.evaluate(() => {
         document.documentElement.classList.add('dark')
         document.documentElement.classList.remove('light')
       })
       await page.waitForTimeout(300) // wait for theme transition
       await page.screenshot({ path: 'e2e/screenshots/landing-full-dark.png', fullPage: true })
     })

     test('showcase card navigates to playground', async ({ page }) => {
       const showcase = page.locator('#use-cases')
       await showcase.scrollIntoViewIfNeeded()
       await page.waitForTimeout(600)
       const firstCard = showcase.locator('a[href*="/playground"]').first()
       await firstCard.click()
       await expect(page).toHaveURL(/\/playground/)
     })

     test('navigation links work', async ({ page }) => {
       // Playground link
       const playgroundLink = page.locator('header a[href="/playground"]').or(page.locator('nav a[href="/playground"]'))
       await expect(playgroundLink.first()).toBeVisible()

       // App link
       const appLink = page.locator('header a[href="/app"]').or(page.locator('nav a[href="/app"]'))
       await expect(appLink.first()).toBeVisible()
     })
   })
   ```

   NOTE: Add `data-testid="showcase-card"` to ShowcaseCard component and `data-testid="feature-card"` to feature cards for reliable test selection. If the components don't have these yet, the test should use more resilient selectors.

1. Create `frontend/e2e/playground.spec.ts`:

   ```typescript
   import { test, expect } from '@playwright/test'

   test.describe('Playground Page', () => {
     test('renders split-pane layout with default movies dataset', async ({ page }) => {
       await page.goto('/playground')
       await page.waitForLoadState('networkidle')

       // Sidebar visible on desktop
       const sidebar = page.locator('aside')
       await expect(sidebar).toBeVisible()

       // Graph canvas visible
       const canvas = page.locator('canvas')
       await expect(canvas).toBeVisible()

       // Connection badge visible
       await expect(page.getByText('Sample Data')).toBeVisible()

       await page.screenshot({ path: 'e2e/screenshots/playground-movies-light.png' })
     })

     test('loads social dataset via URL param', async ({ page }) => {
       await page.goto('/playground?dataset=social')
       await page.waitForLoadState('networkidle')
       await page.waitForTimeout(500) // graph settle time

       await page.screenshot({ path: 'e2e/screenshots/playground-social-light.png' })
     })

     test('loads fraud dataset via URL param', async ({ page }) => {
       await page.goto('/playground?dataset=fraud')
       await page.waitForLoadState('networkidle')
       await page.waitForTimeout(500)

       await page.screenshot({ path: 'e2e/screenshots/playground-fraud-light.png' })
     })

     test('dataset switcher changes graph data', async ({ page }) => {
       await page.goto('/playground')
       await page.waitForLoadState('networkidle')

       // Switch to social dataset
       const switcher = page.locator('select').or(page.locator('[data-testid="dataset-switcher"]'))
       await switcher.selectOption('social')
       await page.waitForTimeout(500)

       // URL should update
       await expect(page).toHaveURL(/dataset=social/)
     })

     test('guided query cards are clickable', async ({ page }) => {
       await page.goto('/playground')
       await page.waitForLoadState('networkidle')

       // Find query cards in sidebar
       const queryCards = page.locator('aside [data-testid="query-card"]').or(page.locator('aside button, aside [role="button"]'))
       const cardCount = await queryCards.count()
       expect(cardCount).toBeGreaterThan(1)

       // Click a non-active card
       if (cardCount > 1) {
         await queryCards.nth(1).click()
         await page.waitForTimeout(300)
       }
     })

     test('stats panel shows counts', async ({ page }) => {
       await page.goto('/playground')
       await page.waitForLoadState('networkidle')

       // Should show node and edge counts somewhere
       await expect(page.getByText(/nodes/i).or(page.getByText(/Nodes/i))).toBeVisible()
     })

     test('dark mode playground', async ({ page }) => {
       await page.goto('/playground')
       await page.waitForLoadState('networkidle')
       await page.evaluate(() => {
         document.documentElement.classList.add('dark')
         document.documentElement.classList.remove('light')
       })
       await page.waitForTimeout(500)
       await page.screenshot({ path: 'e2e/screenshots/playground-movies-dark.png' })
     })
   })
   ```

   NOTE: Add `data-testid="query-card"` to QueryCard and `data-testid="dataset-switcher"` to DatasetSwitcher for reliable selection.
