import { ApiError, type BackendSchemaResponse, type SchemaResponse, type QueryResponse } from '@/types/api'

export class ApiClient {
  constructor(private baseUrl: string) {}

  private async request<T>(path: string, options?: RequestInit): Promise<T> {
    const url = `${this.baseUrl}${path}`
    const res = await fetch(url, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...options?.headers,
      },
    })
    if (!res.ok) {
      const body = await res.json().catch(() => ({ message: res.statusText }))
      throw new ApiError(body.message || res.statusText, res.status, body)
    }
    return res.json()
  }

  async health(): Promise<{ connected: boolean }> {
    try {
      await this.request<unknown>('/health', { signal: AbortSignal.timeout(3000) })
      return { connected: true }
    } catch {
      return { connected: false }
    }
  }

  async query(cypher: string): Promise<QueryResponse> {
    return this.request<QueryResponse>('/query', {
      method: 'POST',
      body: JSON.stringify({ query: cypher }),
    })
  }

  async schema(): Promise<SchemaResponse> {
    const raw = await this.request<BackendSchemaResponse>('/schema')
    return {
      labels: raw.labels ?? [],
      relationshipTypes: raw.edge_types ?? [],
      propertyKeys: raw.property_keys ?? [],
    }
  }
}

```typescript
import type { QueryResponse } from '@/types/api'
import type { BackendQueryResponse } from '@/types/api'
import type { GraphData, GraphNode, GraphEdge } from '@/types/graph'

// Descriptor that tells transformLiveResponse how to reconstruct graph data from rows
export interface NodeColumnDescriptor {
  nameCol: string      // column containing the unique display name
  propsCol: string     // column containing PROPERTIES(n) map
  label: string        // synthetic label for visualization
}

export interface EdgeDescriptor {
  srcCol: string       // column identifying edge source (must match a nameCol)
  dstCol: string       // column identifying edge target (must match a nameCol)
  type: string         // relationship type string
}

export interface GraphQueryDescriptor {
  nodeColumns: NodeColumnDescriptor[]
  edgeDescriptors?: EdgeDescriptor[]
}

// Legacy transform for old {nodes, relationships} format — kept for compatibility
export function transformQueryResponse(response: QueryResponse): GraphData {
  const nodes: GraphNode[] = (response.nodes ?? []).map((n) => ({
    id: n.id,
    labels: n.labels,
    properties: n.properties,
    label: n.labels[0] || String(n.id),
  }))

  const links: GraphEdge[] = (response.relationships ?? []).map((r) => ({
    id: r.id,
    source: r.startNode,
    target: r.endNode,
    type: r.type,
    properties: r.properties,
  }))

  return { nodes, links }
}

// Live transform: converts backend's {columns, rows, row_count} to GraphData
// Uses descriptor to know which columns represent nodes and how to build edges
export function transformLiveResponse(
  response: BackendQueryResponse,
  descriptor: GraphQueryDescriptor
): GraphData {
  const nodeMap = new Map<string, GraphNode>()
  const edgeSet = new Set<string>()
  const links: GraphEdge[] = []

  for (const row of response.rows) {
    // Build nodes from each node column descriptor
    for (const { nameCol, propsCol, label } of descriptor.nodeColumns) {
      const name = row[nameCol]
      if (name == null || name === '') continue
      const nameStr = String(name)
      const key = `${label}:${nameStr}`
      if (!nodeMap.has(key)) {
        const props = (row[propsCol] as Record<string, unknown>) ?? {}
        nodeMap.set(key, {
          id: key,
          labels: [label],
          properties: { ...props, _label: label },
          label,
        })
      }
    }

    // Build edges from edge descriptors
    if (descriptor.edgeDescriptors) {
      for (const { srcCol, dstCol, type } of descriptor.edgeDescriptors) {
        const srcName = row[srcCol]
        const dstName = row[dstCol]
        if (srcName == null || dstName == null) continue
        const srcNameStr = String(srcName)
        const dstNameStr = String(dstName)
        if (!srcNameStr || !dstNameStr) continue

        const srcLabel = descriptor.nodeColumns.find((c) => c.nameCol === srcCol)?.label ?? ''
        const dstLabel = descriptor.nodeColumns.find((c) => c.nameCol === dstCol)?.label ?? ''
        const srcKey = `${srcLabel}:${srcNameStr}`
        const dstKey = `${dstLabel}:${dstNameStr}`
        const edgeId = `${srcKey}--${type}--${dstKey}`

        if (!edgeSet.has(edgeId)) {
          edgeSet.add(edgeId)
          links.push({
            id: edgeId,
            source: srcKey,
            target: dstKey,
            type,
            properties: {},
          })
        }
      }
    }
  }

  return { nodes: Array.from(nodeMap.values()), links }
}
```
